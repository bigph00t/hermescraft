```javascript
// chests.js — Chest inventory tracking for deep memory
import { readFileSync, writeFileSync, existsSync, mkdirSync } from 'fs'
import { join, dirname } from 'path'

let CHESTS_FILE = ''
let chests = {}

export function initChests(agentConfig) {
  CHESTS_FILE = join(agentConfig.dataDir, 'chests.json')
  const dir = dirname(CHESTS_FILE)
  if (!existsSync(dir)) mkdirSync(dir, { recursive: true })
  if (existsSync(CHESTS_FILE)) {
    try { chests = JSON.parse(readFileSync(CHESTS_FILE, 'utf-8')) } catch { chests = {} }
  }
}

export function trackChest(x, y, z, contents) {
  const key = `${x},${y},${z}`
  chests[key] = { contents, updated: new Date().toISOString() }
  saveChests()  // fire-and-forget write
}

export function getChestsForPrompt() {
  const keys = Object.keys(chests)
  if (keys.length === 0) return ''
  const parts = keys.map(key => {
    const items = chests[key].contents.map(i => `${i.item}x${i.count}`).join(', ')
    return `(${key}): ${items}`
  })
  return 'Known chests: ' + parts.join(' | ')
}
```

Create `agent/freestyle.js` with the following structure:

**Header comment:** `// freestyle.js — LLM-designed building plan parser and execution tracker`

**Imports:** `readFileSync, writeFileSync, existsSync, mkdirSync` from `fs`, `join, dirname` from `path`

**Module state:**
- `let _stateFile = ''` — path to `freestyle_state.json`
- `let _contextDir = ''` — path to agent's context directory
- `let _activeFreestyle = null` — current active build: `{ name, planFile, blocks: [{block,x,y,z}], completed: number }`

**`initFreestyle(agentConfig)`:**
- Set `_stateFile = join(agentConfig.dataDir, 'freestyle_state.json')`
- Set `_contextDir = join(agentConfig.dataDir, 'context')`
- If `_stateFile` exists, load it with try/catch (silent catch to empty):
  ```javascript
  try { _activeFreestyle = JSON.parse(readFileSync(_stateFile, 'utf-8')) } catch { _activeFreestyle = null }
  ```
- If `_activeFreestyle` loaded AND the plan file still exists in context dir, log resume message. Otherwise set `_activeFreestyle = null`.

**`parseFreestylePlan(text, originX, originY, originZ)`:**
- Return `null` if `!text`
- Match header: `/^##\s+BUILD:\s*(.+)$/m` — return `null` if not found
- Parse materials section: match `/###\s+Materials\n([\s\S]*?)(?:###|$)/` then line-by-line `/^\s*([a-z_]+):\s*(\d+)/`
- Parse placement section: match `/###\s+Placement\n([\s\S]*?)(?:###|$)/` then line-by-line `/^\s*\d+[.)]\s+([a-z_]+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)/`
- Convert to absolute coords: `x: originX + parseInt(m[2])`, etc.
- If `blocks.length === 0` and placement section existed, log warning: `[freestyle] WARNING: placement section found but 0 blocks parsed — check format`
- Soft cap: if `blocks.length > 200`, log warning and truncate to 200
- Return `{ name, materials, blocks }`

**`startFreestyle(name, planFile, blocks)`:**
- Set `_activeFreestyle = { name, planFile, blocks, completed: 0 }`
- Save state to `_stateFile`

**`advanceFreestyle()`:**
- If no active freestyle, return
- Increment `_activeFreestyle.completed`
- Save state to `_stateFile`
- If completed === blocks.length, call `clearFreestyle()`

**`getNextFreestyleBatch(batchSize = 20)`:**
- If no active freestyle, return `[]`
- Return `_activeFreestyle.blocks.slice(_activeFreestyle.completed, _activeFreestyle.completed + batchSize)`

**`isFreestyleActive()`:**
- Return `!!_activeFreestyle && _activeFreestyle.completed < _activeFreestyle.blocks.length`

**`getFreestyleProgress()`:**
- If no active freestyle, return `null`
- Return `{ name: _activeFreestyle.name, total: _activeFreestyle.blocks.length, completed: _activeFreestyle.completed, remaining: _activeFreestyle.blocks.length - _activeFreestyle.completed }`

**`clearFreestyle()`:**
- Set `_activeFreestyle = null`
- Delete `_stateFile` if it exists (try/catch, silent)

**`_saveFreestyleState()` (internal):**
- Write `_activeFreestyle` to `_stateFile` as JSON
- Fire-and-forget — try/catch with silent catch

**Export all public functions:** `parseFreestylePlan, initFreestyle, startFreestyle, advanceFreestyle, getNextFreestyleBatch, isFreestyleActive, getFreestyleProgress, clearFreestyle`

**Codebase conventions to follow:**
- 2-space indent, single quotes, no semicolons, trailing commas in multiline
- One-line header comment: `// freestyle.js — LLM-designed building plan parser and execution tracker`
- Named exports only, no default export
- camelCase function names
- Module-level state with leading underscore for private (`_activeFreestyle`, `_stateFile`, `_contextDir`)
