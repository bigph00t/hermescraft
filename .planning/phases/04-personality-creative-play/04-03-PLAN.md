// D-10: Parse buildEvaluation from vision context
function parseBuildEvaluation(visionText) {
  if (!visionText) return null
  const match = visionText.match(/^BUILD:\s*(.+)$/m)
  if (!match || match[1].trim().toLowerCase() === 'none') return null
  return match[1].trim()
}
