**File 1: agent/command-parser.js — Add formatLastCommandResults helper**

Add a new exported function at the bottom of the file that formats `_lastCommandResult` entries for planner injection:

```javascript
export function formatLastCommandResults(resultMap) {
  if (!resultMap || resultMap.size === 0) return ''
  const lines = ['Recent command results:']
  for (const [cmd, data] of resultMap) {
    const age = Math.round((Date.now() - data.ts) / 1000)
    if (age > 120) continue  // skip results older than 2 minutes
    lines.push(`  ${cmd}: ${data.summary} (${age}s ago)`)
  }
  return lines.length > 1 ? lines.join('\n') : ''
}
```

This formats the persistent `_lastCommandResult` Map maintained by index.js into a compact string for planner context injection.
