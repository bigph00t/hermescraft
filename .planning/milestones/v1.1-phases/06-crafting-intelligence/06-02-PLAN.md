   ```javascript
   function inventoryToMap(state) {
     const inv = {}
     for (const slot of (state?.inventory || [])) {
       const name = (slot.item || slot.name || '').replace('minecraft:', '')
       if (name) inv[name] = (inv[name] || 0) + (slot.count || 1)
     }
     return inv
   }
   ```
   This converts the state.inventory array format into the `{ itemName: count }` object that solveCraft expects.

4. Inside parseQueueFromPlan, in the `case 'craft':` block (line 103), AFTER the existing `args = { item: parts.slice(1).join('_') }` line, add craft chain expansion:

   ```javascript
   case 'craft': case 'smelt': case 'equip':
     args = { item: parts.slice(1).join('_') }
     // Expand craft chains via crafter.js — resolve dependencies at queue-write time
     if (type === 'craft' && state) {
       try {
         const inv = inventoryToMap(state)
         const chain = solveCraft(args.item, inv)
         if (chain.steps.length > 1) {
           // Replace single craft with full chain
           for (const step of chain.steps) {
             const craftItem = { type: 'craft', args: { item: step.item }, reason: reason || `chain: ${step.item}` }
             // 3x3 recipes need crafting table — two craft calls
             if (step.needsTable) {
               // Check if crafting_table is in inventory; if not, add a craft for it
               if (!inv['crafting_table']) {
                 const tableChain = solveCraft('crafting_table', inv)
                 for (const ts of tableChain.steps) {
                   items.push({ type: 'craft', args: { item: ts.item }, reason: 'need crafting table' })
                 }
                 inv['crafting_table'] = 1 // Track that we'll have one after this
               }
             }
             items.push(craftItem)
           }
           if (items.length >= 20) break
           continue  // Skip the normal push at the end of the for loop
         }
       } catch (e) {
         // Fall through to normal single-craft behavior on error
         console.log('[Planner] Craft chain error:', e.message)
       }
     }
     break
   ```

   IMPORTANT: The `continue` statement skips the `items.push({ type, args, reason })` at line 154, since we already pushed the expanded chain items. For the `smelt` and `equip` cases, flow falls through to the normal push as before.
