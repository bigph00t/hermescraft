**Create server/plugins/Skript/scripts/where.sk:**
```skript
command /where:
    description: Report current coordinates and biome
    trigger:
        set {_x} to round(x-coordinate of player's location)
        set {_y} to round(y-coordinate of player's location)
        set {_z} to round(z-coordinate of player's location)
        set {_biome} to biome of player's location
        send "Location: %{_x}% %{_y}% %{_z}% in %{_biome}%" to player
```
Output format: `Location: 100 64 200 in plains` (single line, parseable by regex).

**Create server/plugins/Skript/scripts/nearbyplayers.sk:**
```skript
command /nearbyplayers [<integer=100>]:
    description: List nearby players with distance and direction
    trigger:
        set {_radius} to arg-1
        if {_radius} > 200:
            set {_radius} to 200
        set {_count} to 0
        loop all players in radius {_radius} of player:
            if loop-player is not player:
                set {_dist} to round(distance between player's location and loop-player's location)
                set {_dx} to (x-coordinate of loop-player's location) - (x-coordinate of player's location)
                set {_dz} to (z-coordinate of loop-player's location) - (z-coordinate of player's location)
                # Determine cardinal direction
                set {_dir} to "nearby"
                if abs({_dx}) >= abs({_dz}):
                    if {_dx} >= 0:
                        set {_dir} to "east"
                    else:
                        set {_dir} to "west"
                else:
                    if {_dz} >= 0:
                        set {_dir} to "south"
                    else:
                        set {_dir} to "north"
                send "Player: %loop-player's name% %{_dist}% blocks %{_dir}%" to player
                add 1 to {_count}
        if {_count} is 0:
            send "No players within %{_radius}% blocks" to player
```
Output format per player: `Player: Steve 45 blocks north` (one line per player). No-match: `No players within 100 blocks`.

**Create server/plugins/Skript/scripts/checkblock.sk:**
```skript
command /checkblock <integer> <integer> <integer>:
    description: Check block type at coordinates
    trigger:
        set {_x} to arg-1
        set {_y} to arg-2
        set {_z} to arg-3
        set {_loc} to location({_x}, {_y}, {_z}, player's world)
        set {_block} to type of block at {_loc}
        send "Block: %{_block}% at %{_x}% %{_y}% %{_z}%" to player
```
Output format: `Block: oak_planks at 100 64 200` (single line). Skript renders block types as lowercase_underscored names matching MC registry.

Add three new typed extractors:

```javascript
// Returns { x, y, z, biome } from where_result.
export function extractWhereResult(parsedResults) {
  const r = parsedResults.find(r => r.type === 'where_result')
  if (!r) return null
  return {
    x: parseInt(r.match[1], 10),
    y: parseInt(r.match[2], 10),
    z: parseInt(r.match[3], 10),
    biome: r.match[4],
  }
}

// Returns array of { name, distance, direction } from nearby_player results.
export function extractNearbyPlayers(parsedResults) {
  return parsedResults
    .filter(r => r.type === 'nearby_player')
    .map(r => ({
      name: r.match[1],
      distance: parseInt(r.match[2], 10),
      direction: r.match[3],
    }))
}

// Returns { block, x, y, z } from check_block result.
export function extractCheckBlockResult(parsedResults) {
  const r = parsedResults.find(r => r.type === 'check_block')
  if (!r) return null
  return {
    block: r.match[1],
    x: parseInt(r.match[2], 10),
    y: parseInt(r.match[3], 10),
    z: parseInt(r.match[4], 10),
  }
}
```
